
apiVersion: apps/v1
kind: Deployment
metadata:
  name: backend
  namespace: lemp-kube
spec:
  replicas: 1
  selector:
    matchLabels:
      app: backend
  template:
    metadata:
      labels:
        app: backend
    spec:
      containers:
      - name: backend
        image: docker.io/library/flask-backend:latest
        imagePullPolicy: IfNotPresent
        ports:
        - containerPort: 5000
        env:
        - name: DB_HOST
          valueFrom:
            secretKeyRef:
              name: db-secret
              key: DB_HOST
        - name: DB_USER
          valueFrom:
            secretKeyRef:
              name: db-secret
              key: DB_USER
        - name: DB_PASS
          valueFrom:
            secretKeyRef:
              name: db-secret
              key: DB_PASS
        - name: DB_NAME
          valueFrom:
            secretKeyRef:
              name: db-secret
              key: DB_NAME

---
apiVersion: v1
kind: Service
metadata:
  name: backend
  namespace: lemp-kube
spec:
  type: NodePort
  selector:
    app: backend
  ports:
    - port: 5000
      nodePort: 30001
